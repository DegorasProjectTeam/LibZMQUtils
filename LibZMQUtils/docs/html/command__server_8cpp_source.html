97void CommandServerBase::startServer()
98{
99 // Safe mutex lock
100 std::unique_lock<std::mutex> lock(this->mtx_);
101
102 // If server is already started, do nothing
103 if (this->server_working_)
104 return;
105
106 // Create the ZMQ context.
107 if (!this->context_)
108 this->context_ = new zmq::context_t(1);
109
110 // Launch server worker in other thread.
111 this->server_worker_future_ = std::async(std::launch::async, &CommandServerBase::serverWorker, this);
112}

143BaseServerResult CommandServerBase::execReqConnect(const CommandRequest& cmd_req)
144{
145 // Safe mutex lock.
146 std::unique_lock<std::mutex> lock(this->mtx_);
147
148 // Check if the client is already connected.
149 auto it = this->connected_clients_.find(cmd_req.client.id);
150 if(it != this->connected_clients_.end())
151 return BaseServerResult::ALREADY_CONNECTED;
152
153 // Add the new client.
154 this->connected_clients_[cmd_req.client.id] = cmd_req.client;
155
156 // Update the timeout of the main socket.
157 if(this->check_clients_alive_)
158 this->updateServerTimeout();
159
160 // Call to the internal callback.
161 this->onConnected(cmd_req.client);
162
163 // All ok.
164 return BaseServerResult::COMMAND_OK;
165}

167BaseServerResult CommandServerBase::execReqDisconnect(const CommandRequest& cmd_req)
168{
169 // Safe mutex lock.
170 std::unique_lock<std::mutex> lock(this->mtx_);
171
172 // Get the client.
173 auto it = this->connected_clients_.find(cmd_req.client.id);
174
175 // Remove the client from the map of connected clients.
176 this->connected_clients_.erase(it);
177
178 // Call to the internal callback.
179 this->onDisconnected(cmd_req.client);
180
181 // Update the timeout of the main socket.
182 if(this->check_clients_alive_)
183 this->updateServerTimeout();
184
185 // All ok.
186 return BaseServerResult::COMMAND_OK;
187}

189void CommandServerBase::serverWorker()
190{
191 // Auxiliar variables.
192 BaseServerResult result;
193
194 // Set the working flag to true.
195 this->server_working_ = true;
196
197 // Start server socket
198 this->resetSocket();
199
200 // Server worker loop.
201 // If there is no client connected wait for a client to connect or for an exit message. If there
202 // is a client connected set timeout, so if no command comes in time, check the last time connection
203 // for each client. The loop can be stopped (in a safe way) if using the stopServer() function.
204 while(this->main_socket_ && this->server_working_)
205 {
206 // Message container.
207 CommandRequest cmd_request;
208
209 // Result container.
210 CommandReply cmd_reply;
211
212 // Receive the data.
213 result = this->recvFromSocket(cmd_request);
214
215 // Check all the clients status.
216 if(this->check_clients_alive_)
217 this->checkClientsAliveStatus();
218
219 // Process the data.
220 if(result == BaseServerResult::COMMAND_OK && !this->server_working_)
221 {
222 // In this case, we will close the server. Call to the internal callback.
223 this->onServerStop();
224 }
225 else if(result == BaseServerResult::TIMEOUT_REACHED)
226 {
227 // DO NOTHING.
228 }
229 else if (result != BaseServerResult::COMMAND_OK)
230 {
231 // Internal callback.
232 this->onInvalidMsgReceived(cmd_request);
233
234 // Prepare the message.
235 std::uint8_t res_buff[sizeof(BaseServerResult)];
236 utils::binarySerializeDeserialize(&result, sizeof(BaseServerResult), res_buff);
237 zmq::message_t message_res(res_buff, sizeof(BaseServerResult));
238
239 // Send response callback.
240 cmd_reply.result = result;
241 this->onSendingResponse(cmd_reply);
242
243 // Send the response.
244 try
245 {
246 this->main_socket_->send(message_res, zmq::send_flags::none);
247 }
248 catch (const zmq::error_t &error)
249 {
250 // Check if we want to close the server.
251 // The error code is for ZMQ EFSM error.
252 if(!(error.num() == common::kZmqEFSMError && !this->server_working_))
253 this->onServerError(error, "Error while sending a response.");
254 }
255 }
256 else if (result == BaseServerResult::COMMAND_OK)
257 {
258 // Reply id buffer.
259 std::unique_ptr<std::uint8_t> rep_id_buff;
260
261 // Execute the command.
262 this->processCommand(cmd_request, cmd_reply);
263
264 // Prepare the command result.
265 CommandServerBase::prepareCommandResult(cmd_reply.result, rep_id_buff);
266 zmq::message_t message_rep_id(rep_id_buff.get(), sizeof(common::CmdReplyRes));
267
268 // Prepare the multipart msg.
269 zmq::multipart_t multipart_msg;
270 multipart_msg.add(std::move(message_rep_id));
271
272 // Specific data.
273 if(cmd_reply.result == BaseServerResult::COMMAND_OK && cmd_reply.params_size != 0)
274 {
275 // Prepare the custom response.
276 zmq::message_t message_rep_custom(cmd_reply.params.get(), cmd_reply.params_size);
277 multipart_msg.add(std::move(message_rep_custom));
278 }
279
280 // Sending callback.
281 this->onSendingResponse(cmd_reply);
282
283 // Send the message.
284 try
285 {
286 multipart_msg.send(*this->main_socket_);
287 }
288 catch (const zmq::error_t &error)
289 {
290 // Check if we want to close the server.
291 // The error code is for ZMQ EFSM error.
292 if(!(error.num() == common::kZmqEFSMError && !this->server_working_))
293 this->onServerError(error, "Error while sending a response.");
294 }
295 }
296 }
297
298 // Delete pointers for clean finish the worker.
299 if (this->main_socket_)
300 {
301 delete this->main_socket_;
302 this->main_socket_ = nullptr;
303 }
304}

306BaseServerResult CommandServerBase::recvFromSocket(CommandRequest& request)
307{
308 // Result variable.
309 BaseServerResult result = BaseServerResult::COMMAND_OK;
310
311 // Containers.
312 bool recv_result;
313 zmq::multipart_t multipart_msg;
314
315 // Try to receive data. If an execption is thrown, receiving fails and an error code is generated.
316 try
317 {
318 // Call to the internal waiting command callback.
319 this->onWaitingCommand();
320
321 // Wait the command.
322 recv_result = multipart_msg.recv(*(this->main_socket_));
323
324 // Store the raw data.
325 request.raw_msg = multipart_msg.clone();
326 }
327 catch(zmq::error_t& error)
328 {
329 // Check if we want to close the server.
330 // The error code is for ZMQ EFSM error.
331 if(error.num() == common::kZmqEFSMError && !this->server_working_)
332 return BaseServerResult::COMMAND_OK;
333
334 // Else, call to error callback.
335 this->onServerError(error, "Error while receiving a request.");
336 return BaseServerResult::INTERNAL_ZMQ_ERROR;
337 }
338
339 // Check for empty msg or timeout reached.
340 if (multipart_msg.empty() && !recv_result)
341 return BaseServerResult::TIMEOUT_REACHED;
342 else if (multipart_msg.empty())
343 return BaseServerResult::EMPTY_MSG;
344
345 // Check the multipart msg size.
346 if (multipart_msg.size() == 4 || multipart_msg.size() == 5)
347 {
348 // Auxiliar containers.
349 std::string ip;
350 std::string hostname;
351 std::string pid;
352
353 // Get the multipart data.
354 zmq::message_t message_ip = multipart_msg.pop();
355 zmq::message_t message_hostname = multipart_msg.pop();
356 zmq::message_t message_pid = multipart_msg.pop();
357 zmq::message_t message_command = multipart_msg.pop();
358
359 // Get the sizes.
360 size_t ip_size_bytes = message_ip.size();
361 size_t host_size_bytes = message_hostname.size();
362 size_t pid_size_bytes = message_pid.size();
363 size_t command_size_bytes = message_command.size();
364
365 // First get the ip data.
366 if (ip_size_bytes > 0)
367 ip = std::string(static_cast<char*>(message_ip.data()), ip_size_bytes);
368 else
369 return BaseServerResult::EMPTY_CLIENT_IP;
370
371 // Get the hostname data.
372 if (host_size_bytes > 0)
373 hostname = std::string(static_cast<char*>(message_hostname.data()), host_size_bytes);
374 else
375 return BaseServerResult::EMPTY_CLIENT_NAME;
376
377 // Get the pid data.
378 if (host_size_bytes > 0)
379 pid = std::string(static_cast<char*>(message_pid.data()), pid_size_bytes);
380 else
381 return BaseServerResult::EMPTY_CLIENT_PID;
382
383 // Update the client info.
384 request.client = HostClient(ip, hostname, pid);
385 request.client.last_connection = std::chrono::steady_clock::now();
386
387 // Update the last connection if the client is connected.
388 this->updateClientLastConnection(request.client.id);
389
390 // Get the command.
391 if (command_size_bytes == sizeof(CmdRequestId))
392 {
393 int raw_command;
394 utils::binarySerializeDeserialize(message_command.data(), sizeof(BaseServerCommand), &raw_command);
395 // Validate the command.
396 if(CommandServerBase::validateCommand(raw_command))
397 request.command = static_cast<BaseServerCommand>(raw_command);
398 else
399 {
400 request.command = BaseServerCommand::INVALID_COMMAND;
401 return BaseServerResult::INVALID_MSG;
402 }
403 }
404 else
405 {
406 request.command = BaseServerCommand::INVALID_COMMAND;
407 return BaseServerResult::INVALID_MSG;
408 }
409
410 // If there is still one more part, they are the parameters.
411 if (multipart_msg.size() == 1)
412 {
413 // Get the message and the size.
414 zmq::message_t message_params = multipart_msg.pop();
415 size_t params_size_bytes = message_params.size();
416
417 std::cout<<multipart_msg.str()<<std::endl;
418 std::cout<<params_size_bytes<<std::endl;
419
420 // Check the parameters.
421 if(params_size_bytes > 0)
422 {
423 // Get and store the parameters data.
424 std::unique_ptr<std::uint8_t> params =
425 std::unique_ptr<std::uint8_t>(new std::uint8_t[params_size_bytes]);
426 auto *params_pointer = static_cast<std::uint8_t*>(message_params.data());
427 std::copy(params_pointer, params_pointer + params_size_bytes, params.get());
428 request.params = std::move(params);
429 request.params_size = params_size_bytes;
430 }
431 else
432 return BaseServerResult::EMPTY_PARAMS;
433 }
434 }
435 else
436 return BaseServerResult::INVALID_PARTS;
437
438 // Return the result.
439 return result;
440}

448bool CommandServerBase::validateCommand(int raw_command)
449{
450 // Auxiliar variables.
451 bool result = false;
452 int reserved_cmd = static_cast<int>(common::BaseServerCommand::RESERVED_COMMANDS);
453 int end_base_cmd = static_cast<int>(common::BaseServerCommand::END_BASE_COMMANDS);
454 // Check if the command is valid.
455 if (raw_command >= common::kMinBaseCmdId && raw_command < reserved_cmd)
456 result = true;
457 else if(raw_command > end_base_cmd)
458 result = true;
459 return result;
460}

462void CommandServerBase::processCommand(const CommandRequest& request, CommandReply& reply)
463{
464 // First of all, call to the internal callback.
465 this->onCommandReceived(request);
466
467 // Store the command in the reply.
468 reply.request_cmd = request.command;
469
470 // Process the different commands.
471 // 1 - Process is the connect request.
472 // 2 - If the command is other, check if the client is connected to the server.
473 // 3 - If it is, check if the command is valid.
474 // 4 - If valid, process the rest of the base commands or the custom command.
475 if (BaseServerCommand::REQ_CONNECT == request.command)
476 {
477 reply.result = this->execReqConnect(request);
478 }
479 else if(this->connected_clients_.find(request.client.id) == this->connected_clients_.end())
480 {
481 reply.result = BaseServerResult::CLIENT_NOT_CONNECTED;
482 }
483 else if (BaseServerCommand::REQ_DISCONNECT == request.command)
484 {
485 reply.result = this->execReqDisconnect(request);
486 }
487 else if (BaseServerCommand::REQ_ALIVE == request.command)
488 {
489 reply.result = BaseServerResult::COMMAND_OK;
490 }
491 else
492 {
493 // Custom command, so call the internal callback.
494 this->onCustomCommandReceived(request, reply);
495
496 // Chek for an invalid msg.
497 if(reply.result == BaseServerResult::INVALID_MSG)
498 this->onInvalidMsgReceived(request);
499 }

502void CommandServerBase::checkClientsAliveStatus()
503{
504 // Safe mutex lock
505 std::unique_lock<std::mutex> lock(this->mtx_);
506
507 // Auxiliar containers.
508 std::vector<std::string> dead_clients;
509 std::chrono::milliseconds timeout(common::kDefaultClientAliveTimeoutMsec);
510 std::chrono::milliseconds min_remaining_time = timeout;
511
512 // Get the current time.
513 utils::SCTimePointStd now = std::chrono::steady_clock::now();
514
515 // Check each connection.
516 for(auto& client : this->connected_clients_)
517 {
518 // Get the last connection time.
519 const auto& last_conn = client.second.last_connection;
520 // Check if the client reaches the timeout checking the last connection time.
521 auto since_last_conn = std::chrono::duration_cast<std::chrono::milliseconds>(now - last_conn);
522 if(since_last_conn >= timeout)
523 {
524 // If dead, call the onDead callback and quit the client from the map.
525 this->onDeadClient(client.second);
526 dead_clients.push_back(client.first);
527 }
528 else
529 {
530 // If the client is not dead, check the minor timeout of the client to set
531 // with the remain time to reach the timeout.
532 min_remaining_time = std::min(min_remaining_time, timeout - since_last_conn);
533 }
534 }
535
536 // Remove dead clients from the map.
537 for(auto& client : dead_clients)
538 {
539 this->connected_clients_.erase(client);
540 }
541
542 // Disable the timeout if no clients remains or set the socket timeout to the
543 // minimum remaining time to the timeout among all clients.
544 if(this->connected_clients_.empty())
545 {
546 this->main_socket_->set(zmq::sockopt::rcvtimeo, -1);
547 }
548 else
549 {
550 this->main_socket_->set(zmq::sockopt::rcvtimeo, static_cast<int>(min_remaining_time.count()));
551 }
552}

564void CommandServerBase::updateServerTimeout()
565{
566 // Calculate the minor timeout to set it into the socket.
567 auto min_timeout = std::min_element(this->connected_clients_.begin(), this->connected_clients_.end(),
568 [](const auto& a, const auto& b)
569 {
570 auto diff_a = std::chrono::duration_cast<std::chrono::milliseconds>(
571 std::chrono::steady_clock::now() - a.second.last_connection);
572 auto diff_b = std::chrono::duration_cast<std::chrono::milliseconds>(
573 std::chrono::steady_clock::now() - b.second.last_connection);
574 return diff_a.count() < diff_b.count();
575 });
576
577 if (min_timeout != this->connected_clients_.end())
578 {
579 auto remain_time = common::kDefaultClientAliveTimeoutMsec - std::chrono::duration_cast<std::chrono::milliseconds>(
580 std::chrono::steady_clock::now() - min_timeout->second.last_connection).count();
581 this->main_socket_->set(zmq::sockopt::rcvtimeo, std::max(0, static_cast<int>(remain_time)));
582 }
583 else
584 {
585 this->main_socket_->set(zmq::sockopt::rcvtimeo, -1);
586 }
587}

589void CommandServerBase::resetSocket()
590{
591 // Auxiliar variables.
592 int res = 0;
593 const zmq::error_t* last_error;
594 unsigned reconnect_count = common::kServerReconnTimes;
595
596 // Delete the previous socket.
597 if (this->main_socket_)
598 {
599 delete this->main_socket_;
600 this->main_socket_ = nullptr;
601 }
602 // Try creating a new socket.
603 do
604 {
605 try
606 {
607 // Create the ZMQ rep socket.
608 std::this_thread::sleep_for(std::chrono::microseconds(500));
609 this->main_socket_ = new zmq::socket_t(*this->context_, zmq::socket_type::rep);
610 this->main_socket_->bind(this->server_endpoint_);
611 this->main_socket_->set(zmq::sockopt::linger, 0);
612 }
613 catch (const zmq::error_t& error)
614 {
615 // Delete the socket and store the last error.
616 delete this->main_socket_;
617 this->main_socket_ = nullptr;
618 last_error = &error;
619 }
620 reconnect_count--;
621 } while (res == EADDRINUSE && reconnect_count > 0);
622
623 if (!this->main_socket_ )
624 {
625 // Update the working flag and calls to the callback.
626 this->server_working_ = false;
627 this->onServerError(*last_error, "Error during socket creation.");
628 }
629 else
630 {
631 // Call to the internal callback.
632 this->onServerStart();
633 }
634}
