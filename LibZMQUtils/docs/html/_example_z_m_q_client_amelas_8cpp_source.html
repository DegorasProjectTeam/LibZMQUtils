39 char *token = std::strtok(command_str, " ");
40
41 if (token)

47 command_id = static_cast<common::CommandType>(std::stoi(token));
48 }

74 valid = false;

76 else if (command_id == static_cast<common::CommandType>(AmelasServerCommand::REQ_SET_DATETIME))
77 {
78 std::cout << "Set datetime command not implemented yet." << std::endl;

82 {
83 std::cout << "Sending get home position command." << std::endl;
84 }
85 else if (command_id == static_cast<common::CommandType>(AmelasServerCommand::REQ_SET_HOME_POSITION))

88
89 bool valid_params = true;
90 double az = 0., el = 0.;
91 char *param_token = std::strtok(nullptr, " ");
92
93 try

95 az = std::stod(param_token);
96 }
97 catch (...)
98 {
99 std::cerr << "Bad parameter azimuth issued.";

105 param_token = std::strtok(nullptr, " ");

131 }

151
152 auto *data_bytes = static_cast<std::uint8_t*>(data_out);
153 zmqutils::utils::binarySerializeDeserialize(

173 }

180 std::cerr << "Command is not implemented or valid" << std::endl;
181 }
182
183 }
184 else
185 {
186 std::cerr << "Not a valid command" << std::endl;
187 }
188
189
190 delete[] command_str;
191}
