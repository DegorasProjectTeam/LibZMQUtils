65std::vector<NetworkAdapterInfo> getHostIPsWithInterfaces()
66{
67 // Result container.
68 std::vector<NetworkAdapterInfo> adapters;
69
70#ifdef _WIN32
71
72 // Buffer size.
73 ULONG buff_size = 0;
74
75 if (GetAdaptersAddresses(AF_INET, GAA_FLAG_SKIP_ANYCAST | GAA_FLAG_SKIP_MULTICAST | GAA_FLAG_SKIP_DNS_SERVER,
76 nullptr, nullptr, &buff_size) != ERROR_BUFFER_OVERFLOW)
77 {
78 return adapters;
79 }
80
81 std::vector<char> buffer(buff_size);
82
83 PIP_ADAPTER_ADDRESSES adapter_addrs = reinterpret_cast<PIP_ADAPTER_ADDRESSES>(&buffer[0]);
84
85 if (GetAdaptersAddresses(AF_INET, GAA_FLAG_SKIP_ANYCAST | GAA_FLAG_SKIP_MULTICAST | GAA_FLAG_SKIP_DNS_SERVER,
86 nullptr, adapter_addrs, &buff_size) != NO_ERROR)
87 {
88 return adapters;
89 }
90
91 while (adapter_addrs != nullptr)
92 {
93 if (adapter_addrs->OperStatus == IfOperStatusUp)
94 {
95 PIP_ADAPTER_UNICAST_ADDRESS unicast_addrs = adapter_addrs->FirstUnicastAddress;
96 while (unicast_addrs != nullptr)
97 {
98 sockaddr_in* sockaddr = reinterpret_cast<sockaddr_in*>(unicast_addrs->Address.lpSockaddr);
99 char* ip = inet_ntoa(sockaddr->sin_addr);
100 char f_name_ch[260];
101 char desc_ch[260];
102 char df_char = ' ';
103
104 WideCharToMultiByte(CP_ACP,0,adapter_addrs->FriendlyName,-1, f_name_ch,260, &df_char, NULL);
105 WideCharToMultiByte(CP_ACP,0,adapter_addrs->Description,-1, desc_ch,260, &df_char, NULL);
106
107 NetworkAdapterInfo adaptr;
108 adaptr.id = std::string(adapter_addrs->AdapterName);
109 adaptr.name = std::string(f_name_ch);
110 adaptr.descr = std::string(desc_ch);
111 adaptr.ip = std::string(ip);
112 adapters.push_back(adaptr);
113 unicast_addrs = unicast_addrs->Next;
114 }
115 }
116
117 adapter_addrs = adapter_addrs->Next;
118 }
119#else
120 // TODO
121#endif
122
123 // Return the ip interface maps.
124 return adapters;
125}

156void binarySerializeDeserialize(const void *data, size_t data_size_bytes, void *dest)
157{
158 const std::uint8_t* data_byes = reinterpret_cast<const std::uint8_t *>(data);
159 std::uint8_t* dest_byes = reinterpret_cast<std::uint8_t*>(dest);
160 std::reverse_copy(data_byes, data_byes + data_size_bytes, dest_byes);
161}

163std::string timePointToString(const HRTimePointStd &tp, const std::string &format, bool add_ms, bool add_ns, bool utc)
164{
165 // Stream to hold the formatted string and the return container.
166 std::ostringstream ss;
167 // Convert the time point to a duration and get the different time fractions.
168 HRTimePointStd::duration dur = tp.time_since_epoch();
169 const time_t secs = duration_cast<std::chrono::seconds>(dur).count();
170 const long long mill = duration_cast<std::chrono::milliseconds>(dur).count();
171 const unsigned long long ns = duration_cast<std::chrono::nanoseconds>(dur).count();
172 const unsigned long long s_ns = secs * 1e9;
173 const unsigned long long t_ns = (ns - s_ns);
174 // Format the duration.
175 if (const std::tm *tm = (utc ? std::gmtime(&secs) : std::localtime(&secs)))
176 {
177 ss << std::put_time(tm, format.c_str());
178 if(add_ms && !add_ns)
179 ss << '.' << std::setw(3) << std::setfill('0') << (mill - secs * 1e3);
180 else if(add_ns)
181 ss << '.' << std::setw(9) << std::setfill('0') << t_ns;
182 }
183 else
184 {
185 // If error, return an empty string.
186 return std::string();
187 }
188 // Return the container.
189 return ss.str();
190}

192std::string timePointToIso8601(const HRTimePointStd &tp, bool add_ms, bool add_ns)
193{
194 // Return the ISO 8601 datetime.
195 return timePointToString(tp, "%Y-%m-%dT%H:%M:%S", add_ms, add_ns) + 'Z';
196}
