116constexpr std::array<T, sizeof...(Is1) + sizeof...(Is2)>
117joinArrays(const std::array<T, sizeof...(Is1)>& a1, const std::array<T, sizeof...(Is2)>& a2, std::index_sequence<Is1...>, std::index_sequence<Is2...>)
