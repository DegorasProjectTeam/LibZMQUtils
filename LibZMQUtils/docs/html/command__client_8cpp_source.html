201int CommandClientBase::recvFromSocket(zmq::socket_t *socket, void *&data, size_t &data_size_bytes) const
202{
203 // Reset output variables
204 data = nullptr;
205 data_size_bytes = 0;
206
207 // Try to receive data. If an execption is thrown, receiving fails and an error code is generated.
208 int result = 0;
209 zmq::recv_result_t recv_result;
210 zmq::message_t message_recv;
211
212 try
213 {
214 recv_result = socket->recv(message_recv);
215 }
216 catch(zmq::error_t& error)
217 {
218 result = error.num();
219 }
220
221 // Return error code if receiving fails or if no data was received
222 if (!recv_result.has_value())
223 {
224 result = EAGAIN;
225 }
226 else if (0 == result)
227 {
228 // If data was received, copy received data to out parameters and return not error code
229 data_size_bytes = message_recv.size();
230 if (data_size_bytes > 0)
231 {
232 uint8_t *msg_data = new uint8_t[data_size_bytes];
233 uint8_t *message_recv_pointer = static_cast<uint8_t*>(message_recv.data());
234 std::copy(message_recv_pointer, message_recv_pointer + data_size_bytes, msg_data);
235 data = msg_data;
236 }
237 }
238
239 return result;
240}

242void CommandClientBase::sendAliveCallback()
243{
244 std::mutex m;
245 std::unique_lock<std::mutex> lk(m);
246 bool send_success = true;
247 bool recv_success = true;
248 void *data_out;
249 size_t out_size;
250 zmq::multipart_t msg;
251 zmq::socket_t *alive_socket = new zmq::socket_t(*this->context_, zmq::socket_type::req);
252 alive_socket->connect(this->server_endpoint_);
253 // Set timeout so socket will not wait for answer more than client alive timeout.
254 alive_socket->set(zmq::sockopt::rcvtimeo, CommandClientBase::kClientAliveTimeoutMsec);
255 alive_socket->set(zmq::sockopt::linger, 0);
256
257 while(this->auto_alive_working_)
258 {
259 auto res =
260 this->auto_alive_cv_.wait_for(lk, std::chrono::milliseconds(CommandClientBase::kClientSendAlivePeriodMsec));
261
262 if (std::cv_status::timeout == res)
263 {
264 msg = this->prepareMessage(
265 CommandData(static_cast<common::CommandType>(BaseServerCommand::REQ_ALIVE)));
266 try
267 {
268 msg.send(*alive_socket);
269 } catch (const zmq::error_t &error)
270 {
271 // TODO: handle error
272 std::cerr << "Failed to automatically send alive command with error: " << error.num() << std::endl;
273 send_success = false;
274 }
275
276 if (send_success)
277 {
278 auto recv_result = this->recvFromSocket(alive_socket, data_out, out_size);
279 auto *data_bytes = static_cast<std::uint8_t*>(data_out);
280
281 if (0 == recv_result && out_size == sizeof(CommandClientBase::CommandError))
282 {
283 CommandClientBase::CommandError error;
284
285
286 zmqutils::utils::binarySerializeDeserialize(
287 data_bytes, sizeof(CommandClientBase::CommandError), &error);
288
289 recv_success = error == CommandClientBase::CommandError::NOT_ERROR;
290
291 }
292 else
293 {
294 std::cerr << "Auto alive message answer receive failed" << std::endl;
295 recv_success = false;
296 }
297
298 delete[] data_bytes;
299 }
300
301 if (!send_success || !recv_success)
302 {
303 std::cerr << "Failed auto sending alive message. Process will be stopped." << std::endl;
304 this->auto_alive_working_ = false;
305 }
306
307 }
308
309 }
310
311 delete alive_socket;
312}

314zmq::multipart_t CommandClientBase::prepareMessage(const CommandData &msg)
315{
316 // Prepare the ip data.
317 zmq::message_t message_ip(this->client_info_.ip.begin(), this->client_info_.ip.end());
318 // Prepare the hostname data.
319 zmq::message_t message_host(this->client_info_.hostname.begin(), this->client_info_.hostname.end());
320 // Prepare the pid data.
321 zmq::message_t message_pid(this->client_info_.pid.begin(), this->client_info_.pid.end());
322 // Prepare the command data.
323 std::uint8_t command_buffer[sizeof(common::CmdRequestId)];
324 zmqutils::utils::binarySerializeDeserialize(&msg.command_id, sizeof(common::CmdRequestId), command_buffer);
325 zmq::message_t message_command(&command_buffer, sizeof(common::CmdRequestId));
326
327
328 // Prepare the multipart msg.
329 zmq::multipart_t multipart_msg;
330 multipart_msg.add(std::move(message_ip));
331 multipart_msg.add(std::move(message_host));
332 multipart_msg.add(std::move(message_pid));
333 multipart_msg.add(std::move(message_command));
334
335 // Add command parameters if they exist
336 if (msg.params_size > 0)
337 {
338 // Prepare the command parameters
339 zmq::message_t message_params(msg.params.get(), msg.params_size);
340 multipart_msg.add(std::move(message_params));
341 }
342
343 return multipart_msg;
344}
